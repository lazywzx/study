\section{表达式与方程组}

大括号
\begin{equation}
	\omega(u, \alpha) = 
	\left\{
		\begin{aligned}
			\tan(\frac{\pi \alpha}{2}), &\quad \alpha \neq 1 \\
			\frac{2}{\pi}\log(|u|), & \quad \alpha = 1
		\end{aligned}
	\right.
\end{equation}

\begin{equation}
	sgn(u) = 
	\left\{
		\begin{aligned}
			1, & \quad u > 0 \\
			0, & \quad u = 0 \\
			-1, & \quad u < 0
		\end{aligned}
	\right.
\end{equation}

离散求和
\begin{align}
    H(x) = \sum_{i=0, j=0}^{N-1, M-1}(x_{i, j} = x), x = 0, 1, \cdots, 255
\end{align}

离散求和（方程组）
\begin{align}
    u &= \sum_{i=0}^{k}\sum_{j=0}^{i} a_{ij} \cdot x^{i}y^{i-j} \\
    v &= \sum_{i=0}^{k}\sum_{j=0}^{i} b_{ij} \cdot x^{i}y^{i-j}
\end{align}

均方根误差RMSE
\begin{align}
    RMSE_{x} = \sqrt{\frac{\sum\limits_{i=0}^{N-1} {(x_{i}^{'} - u_{i})}^{2}}{N}} \\
    RMSE_{y} = \sqrt{\frac{\sum\limits_{i=0}^{N-1} {(y_{i}^{'} - v_{i})}^{2}}{N}}
\end{align}

\section{向量与矩阵}

矩阵表达(斜省略号)
\begin{equation}
    {\sum}_{y}
    = 
    \begin{bmatrix}
        \lambda_{1} & 0 & & \cdots & 0 \\
        0 & \lambda_{2} & & &  \\
         & & \ddots & & \vdots \\
        \vdots & & & \ddots & \\
        0 & &\cdots & &\lambda_{B} \\
    \end{bmatrix}
    ,
    \lambda_{1} > \lambda_{2} > \cdots > \lambda_{B}
\end{equation}

矩阵运算
\begin{equation}
	\begin{bmatrix}
		u \\
		v
	\end{bmatrix}
	= 
	\begin{bmatrix}
		a_{0} & a_{1} & a_{2} \\
		b_{0} & b_{1} & b_{2}
	\end{bmatrix}
	\cdot
	\begin{bmatrix}
		1 & x & y
	\end{bmatrix}
	^{T}
\end{equation}
